import { createAdSlot, placeBid } from 'castflow-sdk';

// Create an ad slot on your cast
const slot = await createAdSlot({
  castHash: '0x123...',
  reservePrice: 10, // USDC
  expiresAt: '2026-07-15'
});

// Place a bid
await placeBid({
  slotId: slot.id,
  amount: 15 // USDC
});
